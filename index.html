const util = require('util');
const colors = {
    reset: "\x1b[0m",
    bright: "\x1b[1m",
    dim: "\x1b[2m",
    underscore: "\x1b[4m",
    reverse: "\x1b[7m",
    fg: {
        black: "\x1b[30m",
        red: "\x1b[31m",
        green: "\x1b[32m",
        yellow: "\x1b[33m",
        blue: "\x1b[34m",
        magenta: "\x1b[35m",
        cyan: "\x1b[36m",
        white: "\x1b[37m"
    },
    bg: {
        black: "\x1b[40m",
        red: "\x1b[41m",
        green: "\x1b[42m",
        yellow: "\x1b[43m",
        blue: "\x1b[44m",
        magenta: "\x1b[45m",
        cyan: "\x1b[46m",
        white: "\x1b[47m"
    }
};

class Logger {
    static formatDate() {
        const now = new Date();
        return `${now.toLocaleDateString()} ${now.toLocaleTimeString()}`;
    }

    static log(level, message, color) {
        const formattedDate = this.formatDate();
        const formattedMessage = util.format(
            '%s[%s]%s %s%s%s',
            color, formattedDate, colors.reset, colors.fg.white, message, colors.reset
        );
        console.log(formattedMessage);
    }

    static info(message) {
        this.log('INFO', message, colors.fg.cyan);
    }

    static warn(message) {
        this.log('WARN', message, colors.fg.yellow);
    }

    static error(message) {
        this.log('ERROR', message, colors.fg.red);
    }

    static debug(message) {
        this.log('DEBUG', message, colors.fg.magenta);
    }

    static success(message) {
        this.log('SUCCESS', message, colors.fg.green);
    }
}

module.exports = Logger;
